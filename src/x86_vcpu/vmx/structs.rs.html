11/// VMCS/VMXON region in 4K size. (SDM Vol. 3C, Section 24.2)
12#[derive(Debug)]
13pub struct VmxRegion<H: AxMmHal> {
14    frame: PhysFrame<H>,
15}
16
17impl<H: AxMmHal> VmxRegion<H> {
18    pub const unsafe fn uninit() -> Self {
19        Self {
20            frame: unsafe { PhysFrame::uninit() },
21        }
22    }
23
24    pub fn new(revision_id: u32, shadow_indicator: bool) -> AxResult<Self> {
25        let frame = PhysFrame::alloc_zero()?;
26        unsafe {
27            (*(frame.as_mut_ptr() as *mut u32))
28                .set_bits(0..=30, revision_id)
29                .set_bit(31, shadow_indicator);
30        }
31        Ok(Self { frame })
32    }
33
34    pub fn phys_addr(&self) -> HostPhysAddr {
35        self.frame.start_paddr()
36    }
37}

39// (SDM Vol. 3C, Section 25.6.4)
40// The VM-execution control fields include the 64-bit physical addresses of I/O bitmaps A and B (each of which are 4 KBytes in size).
41// I/O bitmap A contains one bit for each I/O port in the range 0000H through 7FFFH;
42// I/O bitmap B contains bits for ports in the range 8000H through FFFFH.
43#[derive(Debug)]
44pub struct IOBitmap<H: AxMmHal> {
45    io_bitmap_a_frame: PhysFrame<H>,
46    io_bitmap_b_frame: PhysFrame<H>,
47}
48
49impl<H: AxMmHal> IOBitmap<H> {
50    pub fn passthrough_all() -> AxResult<Self> {
51        Ok(Self {
52            io_bitmap_a_frame: PhysFrame::alloc_zero()?,
53            io_bitmap_b_frame: PhysFrame::alloc_zero()?,
54        })
55    }
56
57    #[allow(unused)]
58    pub fn intercept_all() -> AxResult<Self> {
59        let mut io_bitmap_a_frame = PhysFrame::alloc()?;
60        io_bitmap_a_frame.fill(u8::MAX);
61        let mut io_bitmap_b_frame = PhysFrame::alloc()?;
62        io_bitmap_b_frame.fill(u8::MAX);
63        Ok(Self {
64            io_bitmap_a_frame,
65            io_bitmap_b_frame,
66        })
67    }
68
69    pub fn phys_addr(&self) -> (HostPhysAddr, HostPhysAddr) {
70        (
71            self.io_bitmap_a_frame.start_paddr(),
72            self.io_bitmap_b_frame.start_paddr(),
73        )
74    }
75
76    // Execution of an I/O instruction causes a VM exit
77    // if any bit in the I/O bitmaps corresponding to a port it accesses is 1.
78    // See Section 26.1.3 for details.
79    pub fn set_intercept(&mut self, port: u32, intercept: bool) {
80        let (port, io_bit_map_frame) = if port <= 0x7fff {
81            (port, &mut self.io_bitmap_a_frame)
82        } else {
83            (port - 0x8000, &mut self.io_bitmap_b_frame)
84        };
85        let bitmap =
86            unsafe { core::slice::from_raw_parts_mut(io_bit_map_frame.as_mut_ptr(), 1024) };
87        let byte = (port / 8) as usize;
88        let bits = port % 8;
89        if intercept {
90            bitmap[byte] |= 1 << bits;
91        } else {
92            bitmap[byte] &= !(1 << bits);
93        }
94    }
95
96    pub fn set_intercept_of_range(&mut self, port_base: u32, count: u32, intercept: bool) {
97        for port in port_base..port_base + count {
98            self.set_intercept(port, intercept)
99        }
100    }
101}

103#[derive(Debug)]
104pub struct MsrBitmap<H: AxMmHal> {
105    frame: PhysFrame<H>,
106}
107
108impl<H: AxMmHal> MsrBitmap<H> {
109    pub fn passthrough_all() -> AxResult<Self> {
110        Ok(Self {
111            frame: PhysFrame::alloc_zero()?,
112        })
113    }
114
115    #[allow(unused)]
116    pub fn intercept_all() -> AxResult<Self> {
117        let mut frame = PhysFrame::alloc()?;
118        frame.fill(u8::MAX);
119        Ok(Self { frame })
120    }
121
122    pub fn phys_addr(&self) -> HostPhysAddr {
123        self.frame.start_paddr()
124    }
125
126    fn set_intercept(&mut self, msr: u32, is_write: bool, intercept: bool) {
127        let offset = if msr <= 0x1fff {
128            if !is_write {
129                0 // Read bitmap for low MSRs (0x0000_0000..0x0000_1FFF)
130            } else {
131                2 // Write bitmap for low MSRs (0x0000_0000..0x0000_1FFF)
132            }
133        } else if (0xc000_0000..=0xc000_1fff).contains(&msr) {
134            if !is_write {
135                1 // Read bitmap for high MSRs (0xC000_0000..0xC000_1FFF)
136            } else {
137                3 // Write bitmap for high MSRs (0xC000_0000..0xC000_1FFF)
138            }
139        } else {
140            unreachable!()
141        } * 1024;
142        let bitmap =
143            unsafe { core::slice::from_raw_parts_mut(self.frame.as_mut_ptr().add(offset), 1024) };
144        let msr = msr & 0x1fff;
145        let byte = (msr / 8) as usize;
146        let bits = msr % 8;
147        if intercept {
148            bitmap[byte] |= 1 << bits;
149        } else {
150            bitmap[byte] &= !(1 << bits);
151        }
152    }
153
154    pub fn set_read_intercept(&mut self, msr: u32, intercept: bool) {
155        self.set_intercept(msr, false, intercept);
156    }
157
158    pub fn set_write_intercept(&mut self, msr: u32, intercept: bool) {
159        self.set_intercept(msr, true, intercept);
160    }
161}

163/// Reporting Register of Basic VMX Capabilities. (SDM Vol. 3D, Appendix A.1)
164#[derive(Debug)]
165pub struct VmxBasic {
166    /// The 31-bit VMCS revision identifier used by the processor.
167    pub revision_id: u32,
168    /// The number of bytes that software should allocate for the VMXON region
169    /// and any VMCS region.
170    pub region_size: u16,
171    /// The width of the physical addresses that may be used for the VMXON
172    /// region, each VMCS, and data structures referenced by pointers in a VMCS.
173    pub is_32bit_address: bool,
174    /// The memory type that should be used for the VMCS, for data structures
175    /// referenced by pointers in the VMCS.
176    pub mem_type: u8,
177    /// The processor reports information in the VM-exit instruction-information
178    /// field on VM exits due to execution of the INS and OUTS instructions.
179    pub io_exit_info: bool,
180    /// If any VMX controls that default to 1 may be cleared to 0.
181    pub vmx_flex_controls: bool,
182}
183
184impl MsrReadWrite for VmxBasic {
185    const MSR: Msr = Msr::IA32_VMX_BASIC;
186}
187
188impl VmxBasic {
189    pub const VMX_MEMORY_TYPE_WRITE_BACK: u8 = 6;
190
191    /// Read the current IA32_VMX_BASIC flags.
192    pub fn read() -> Self {
193        let msr = Self::read_raw();
194        Self {
195            revision_id: msr.get_bits(0..31) as u32,
196            region_size: msr.get_bits(32..45) as u16,
197            is_32bit_address: msr.get_bit(48),
198            mem_type: msr.get_bits(50..54) as u8,
199            io_exit_info: msr.get_bit(54),
200            vmx_flex_controls: msr.get_bit(55),
201        }
202    }
203}

206    /// IA32_FEATURE_CONTROL flags.
207    #[derive(Debug)]
208    pub struct FeatureControlFlags: u64 {
209       /// Lock bit: when set, locks this MSR from being written. when clear,
210       /// VMXON causes a #GP.
211       const LOCKED = 1 << 0;
212       /// Enable VMX inside SMX operation.
213       const VMXON_ENABLED_INSIDE_SMX = 1 << 1;
214       /// Enable VMX outside SMX operation.
215       const VMXON_ENABLED_OUTSIDE_SMX = 1 << 2;
216   }
217}

219/// Control Features in Intel 64 Processor. (SDM Vol. 3C, Section 23.7)
220pub struct FeatureControl;
221
222impl MsrReadWrite for FeatureControl {
223    const MSR: Msr = Msr::IA32_FEATURE_CONTROL;
224}
225
226impl FeatureControl {
227    /// Read the current IA32_FEATURE_CONTROL flags.
228    pub fn read() -> FeatureControlFlags {
229        FeatureControlFlags::from_bits_truncate(Self::read_raw())
230    }
231
232    /// Write IA32_FEATURE_CONTROL flags, preserving reserved values.
233    pub fn write(flags: FeatureControlFlags) {
234        let old_value = Self::read_raw();
235        let reserved = old_value & !(FeatureControlFlags::all().bits());
236        let new_value = reserved | flags.bits();
237        unsafe { Self::write_raw(new_value) };
238    }
239}

241bitflags! {
242    /// Extended-Page-Table Pointer. (SDM Vol. 3C, Section 24.6.11)
243    #[derive(Debug)]
244    pub struct EPTPointer: u64 {
245        /// EPT paging-structure memory type: Uncacheable (UC).
246        #[allow(clippy::identity_op)]
247        const MEM_TYPE_UC = 0 << 0;
248        /// EPT paging-structure memory type: Write-back (WB).
249        #[allow(clippy::identity_op)]
250        const MEM_TYPE_WB = 6 << 0;
251        /// EPT page-walk length 1.
252        const WALK_LENGTH_1 = 0 << 3;
253        /// EPT page-walk length 2.
254        const WALK_LENGTH_2 = 1 << 3;
255        /// EPT page-walk length 3.
256        const WALK_LENGTH_3 = 2 << 3;
257        /// EPT page-walk length 4.
258        const WALK_LENGTH_4 = 3 << 3;
259        /// Setting this control to 1 enables accessed and dirty flags for EPT.
260        const ENABLE_ACCESSED_DIRTY = 1 << 6;
261    }
262}
263
264impl EPTPointer {
265    pub fn from_table_phys(pml4_paddr: HostPhysAddr) -> Self {
266        let aligned_addr = pml4_paddr.as_usize() & !(PAGE_SIZE - 1);
267        let flags = Self::from_bits_retain(aligned_addr as u64);
268        flags | Self::MEM_TYPE_WB | Self::WALK_LENGTH_4 | Self::ENABLE_ACCESSED_DIRTY
269    }
270}
