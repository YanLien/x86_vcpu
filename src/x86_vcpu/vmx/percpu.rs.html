12/// Represents the per-CPU state for Virtual Machine Extensions (VMX).
13///
14/// This structure holds the state information specific to a CPU core
15/// when operating in VMX mode, including the VMCS revision identifier and
16/// the VMX region.
17#[derive(Debug)]
18pub struct VmxPerCpuState<H: AxVCpuHal> {
19    /// The VMCS (Virtual Machine Control Structure) revision identifier.
20    ///
21    /// This identifier is used to ensure compatibility between the software
22    /// and the specific version of the VMCS that the CPU supports.
23    pub(crate) vmcs_revision_id: u32,
24
25    /// The VMX region for this CPU.
26    ///
27    /// This region typically contains the VMCS and other state information
28    /// required for managing virtual machines on this particular CPU.
29    vmx_region: VmxRegion<H::MmHal>,
30}
31
32impl<H: AxVCpuHal> AxArchPerCpu for VmxPerCpuState<H> {
33    fn new(_cpu_id: usize) -> AxResult<Self> {
34        Ok(Self {
35            vmcs_revision_id: 0,
36            vmx_region: unsafe { VmxRegion::uninit() },
37        })
38    }
39
40    fn is_enabled(&self) -> bool {
41        Cr4::read().contains(Cr4Flags::VIRTUAL_MACHINE_EXTENSIONS)
42    }
43
44    fn hardware_enable(&mut self) -> AxResult {
45        if !has_hardware_support() {
46            return ax_err!(Unsupported, "CPU does not support feature VMX");
47        }
48        if self.is_enabled() {
49            return ax_err!(ResourceBusy, "VMX is already turned on");
50        }
51
52        // Enable XSAVE/XRSTOR.
53        super::vcpu::XState::enable_xsave();
54
55        // Enable VMXON, if required.
56        let ctrl = FeatureControl::read();
57        let locked = ctrl.contains(FeatureControlFlags::LOCKED);
58        let vmxon_outside = ctrl.contains(FeatureControlFlags::VMXON_ENABLED_OUTSIDE_SMX);
59        if !locked {
60            FeatureControl::write(
61                ctrl | FeatureControlFlags::LOCKED | FeatureControlFlags::VMXON_ENABLED_OUTSIDE_SMX,
62            )
63        } else if !vmxon_outside {
64            return ax_err!(Unsupported, "VMX disabled by BIOS");
65        }
66
67        // Check control registers are in a VMX-friendly state. (SDM Vol. 3C, Appendix A.7, A.8)
68        macro_rules! cr_is_valid {
69            ($value: expr, $crx: ident) => {{
70                use Msr::*;
71                let value = $value;
72                paste::paste! {
73                    let fixed0 = [<IA32_VMX_ $crx _FIXED0>].read();
74                    let fixed1 = [<IA32_VMX_ $crx _FIXED1>].read();
75                }
76                (!fixed0 | value != 0) && (fixed1 | !value != 0)
77            }};
78        }
79        if !cr_is_valid!(Cr0::read().bits(), CR0) {
80            return ax_err!(BadState, "host CR0 is not valid in VMX operation");
81        }
82        if !cr_is_valid!(Cr4::read().bits(), CR4) {
83            return ax_err!(BadState, "host CR4 is not valid in VMX operation");
84        }
85
86        // Get VMCS revision identifier in IA32_VMX_BASIC MSR.
87        let vmx_basic = VmxBasic::read();
88        if vmx_basic.region_size as usize != PAGE_SIZE {
89            return ax_err!(Unsupported);
90        }
91        if vmx_basic.mem_type != VmxBasic::VMX_MEMORY_TYPE_WRITE_BACK {
92            return ax_err!(Unsupported);
93        }
94        if vmx_basic.is_32bit_address {
95            return ax_err!(Unsupported);
96        }
97        if !vmx_basic.io_exit_info {
98            return ax_err!(Unsupported);
99        }
100        if !vmx_basic.vmx_flex_controls {
101            return ax_err!(Unsupported);
102        }
103        self.vmcs_revision_id = vmx_basic.revision_id;
104        self.vmx_region = VmxRegion::new(self.vmcs_revision_id, false)?;
105
106        unsafe {
107            // Enable VMX using the VMXE bit.
108            Cr4::write(Cr4::read() | Cr4Flags::VIRTUAL_MACHINE_EXTENSIONS);
109            // Execute VMXON.
110            vmx::vmxon(self.vmx_region.phys_addr().as_usize() as _).map_err(|err| {
111                ax_err_type!(
112                    BadState,
113                    format_args!("VMX instruction vmxon failed: {:?}", err)
114                )
115            })?;
116        }
117        info!("[AxVM] succeeded to turn on VMX.");
118
119        Ok(())
120    }
121
122    fn hardware_disable(&mut self) -> AxResult {
123        if !self.is_enabled() {
124            return ax_err!(BadState, "VMX is not enabled");
125        }
126
127        unsafe {
128            // Execute VMXOFF.
129            vmx::vmxoff().map_err(|err| {
130                ax_err_type!(
131                    BadState,
132                    format_args!("VMX instruction vmxoff failed: {:?}", err)
133                )
134            })?;
135            // Remove VMXE bit in CR4.
136            Cr4::update(|cr4| cr4.remove(Cr4Flags::VIRTUAL_MACHINE_EXTENSIONS));
137        };
138        info!("[AxVM] succeeded to turn off VMX.");
139
140        self.vmx_region = unsafe { VmxRegion::uninit() };
141        Ok(())
142    }
143}

175    #[test]
176    fn test_multiple_cpu_states_independence() {
177        MockMmHal::reset(); // Reset before test
178        let mut states = Vec::new();
179
180        // Create states for multiple CPUs
181        for cpu_id in 0..4 {
182            let state = VmxPerCpuState::<MockVCpuHal>::new(cpu_id).unwrap();
183            states.push(state);
184        }
185
186        // Test independence by modifying one state and verifying others are unaffected
187        states[0].vmcs_revision_id = 0x12345678;
188        states[1].vmcs_revision_id = 0x87654321;
189
190        // Verify each state maintains its own value
191        assert_eq!(states[0].vmcs_revision_id, 0x12345678);
192        assert_eq!(states[1].vmcs_revision_id, 0x87654321);
193        assert_eq!(states[2].vmcs_revision_id, 0);
194        assert_eq!(states[3].vmcs_revision_id, 0);
195    }
